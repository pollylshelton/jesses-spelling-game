        function renderChoices() {

            const container = document.getElementById('multi-choice'); if (!container) return; container.innerHTML = '';

            const target = window.state.currentWord;

            let opts = [target];

            

            if (window.state.currentGame === 'mermaid') {

                const anchor = phonicAnchors[target];

                const anchorWord = anchor ? anchor.word.toLowerCase() : "";

                // Rule: Distractors must not exist in the anchor word

                const distractors = phonicsList.filter(p => !anchorWord.includes(p) && p !== target);

                while(opts.length < 4) {

                    let r = distractors[Math.floor(Math.random()*distractors.length)];

                    if(!opts.includes(r)) opts.push(r);

                }

            } else if (window.state.currentGame === 'unicorn') {

                const pool = Object.keys(illusts);

                while(opts.length < 3) { let r = pool[Math.floor(Math.random()*pool.length)]; if(!opts.includes(r)) opts.push(r); }

            } else {

                while(opts.length < 4) { let r = wordsDB.recep[Math.floor(Math.random()*wordsDB.recep.length)]; if(!opts.includes(r)) opts.push(r); }

            }

            opts.sort(() => Math.random() - 0.5).forEach(o => {

                const b = document.createElement('button');

                if (window.state.currentGame === 'unicorn') { b.className = 'illustration-card text-6xl drop-shadow-xl'; b.innerHTML = illusts[o] || `❔`; }

                else { b.className = (window.state.currentGame === 'mermaid' ? 'bubble-btn text-white' : 'p-6 bg-white/10 rounded-3xl text-2xl font-bold'); b.innerText = window.toSentenceCase(o); }

                b.onclick = () => {

                    if (window.state.isTransitioning) return;

                    window.state.isTransitioning = true;

                    if (o === target) { recordResult(true); nextRound(); }

                    else { 

                        window.state.lives--; 

                        if (window.state.lives <= 0) { recordResult(false, o); triggerExplosion(nextRound); } 

                        else { triggerExplosion(() => { window.state.isTransitioning = false; }); } 

                    }

                };

                container.appendChild(b);

            });

        }


        function renderMeteorBlast() {

            const field = document.getElementById('meteor-field'); if (!field) return; field.innerHTML = '';

            let opts = [window.state.currentWord];

            while(opts.length < 4) { let r = wordsDB.recep[Math.floor(Math.random()*wordsDB.recep.length)]; if(!opts.includes(r)) opts.push(r); }

            opts.sort(() => Math.random() - 0.5).forEach((text, i) => {

                const m = document.createElement('div'); m.className = 'meteor'; m.innerText = window.toSentenceCase(text);

                let x = -200, y = 30 + (i * 90), speed = 1.1 + (Math.random() * 1.2);

                m.style.top = y + 'px'; field.appendChild(m);

                let processed = false;

                const move = () => {

                    if (processed || !document.body.classList.contains('meteor-active')) return;

                    x += speed; m.style.left = x + 'px';

                    if (x > field.offsetWidth + 200) {

                        processed = true;

                        if (text === window.state.currentWord && !window.state.isTransitioning) {

                            window.state.isTransitioning = true; window.state.lives--; recordResult(false, "Missed"); triggerExplosion(nextRound);

                        }

                    } else { const animId = requestAnimationFrame(move); window.state.meteorTimers.push(animId); }

                };

                requestAnimationFrame(move);

                m.onclick = () => {

                    if (processed || window.state.isTransitioning) return;

                    processed = true; m.style.opacity = '0'; window.state.isTransitioning = true;

                    if (text === window.state.currentWord) { recordResult(true); nextRound(); }

                    else { window.state.lives--; if (window.state.lives <= 0) { recordResult(false, text); triggerExplosion(nextRound); } else { triggerExplosion(() => { window.state.isTransitioning = false; }); } }

                };

            });

        }
